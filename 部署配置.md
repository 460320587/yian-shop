      - ./:/var/www/html:cached

## 2. Dockerfile

### 2.1 多阶段构建 Dockerfile

```dockerfile
# ============================================================
# 阶段1: Builder (安装依赖)
# ============================================================
FROM php:8.3-fpm-alpine AS builder

WORKDIR /var/www/html

# 安装系统依赖
RUN apk add --no-cache \
    git \
    unzip \
    curl \
    libzip-dev \
    libpng-dev \
    libjpeg-turbo-dev \
    freetype-dev \
    oniguruma-dev \
    libxml2-dev \
    linux-headers

# 安装PHP扩展
RUN docker-php-ext-configure gd --with-freetype --with-jpeg \
    && docker-php-ext-install -j$(nproc) \
        pdo_mysql \
        mbstring \
        zip \
        exif \
        pcntl \
        bcmath \
        gd \
        opcache

# 安装Composer
COPY --from=composer:latest /usr/bin/composer /usr/bin/composer

# 复制依赖文件并安装
COPY composer.json composer.lock ./
RUN composer install --no-dev --no-scripts --no-autoloader --prefer-dist

# 复制项目文件
COPY . .

# 生成自动加载并运行脚本
RUN composer dump-autoload --optimize \
    && composer run-script post-autoload-dump

# ============================================================
# 阶段2: Runtime (运行环境)
# ============================================================
FROM php:8.3-fpm-alpine AS runtime

WORKDIR /var/www/html

# 安装运行时依赖
RUN apk add --no-cache \
    libzip \
    libpng \
    libjpeg-turbo \
    freetype \
    oniguruma \
    libxml2 \
    curl \
    supervisor

# 安装PHP扩展
RUN docker-php-ext-install -j$(nproc) \
    pdo_mysql \
    mbstring \
    zip \
    exif \
    pcntl \
    bcmath \
    gd \
    opcache

# 安装Swoole (Octane用)
RUN apk add --no-cache --virtual .build-deps \
    $PHPIZE_DEPS \
    linux-headers \
    && pecl install swoole-5.1.2 \
    && docker-php-ext-enable swoole \
    && apk del .build-deps

# 复制项目文件
COPY --from=builder /var/www/html /var/www/html

# 设置权限
RUN chown -R www-data:www-data /var/www/html \
    && chmod -R 755 /var/www/html/storage \
    && chmod -R 755 /var/www/html/bootstrap/cache

# 健康检查
HEALTHCHECK --interval=30s --timeout=10s --start-period=40s --retries=3 \
    CMD php artisan octane:status || exit 1

EXPOSE 8000

USER www-data

CMD ["php", "artisan", "octane:start", "--server=swoole", "--host=0.0.0.0", "--port=8000"]
```

## 3. Nginx 配置

### 3.1 生产环境 nginx.conf

```nginx
upstream octane {
    server app:8000 weight=5 max_fails=3 fail_timeout=30s;
    keepalive 64;
}

server {
    listen 80;
    server_name www.yian.com api.yian.com;
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name www.yian.com;

    # SSL证书
    ssl_certificate /etc/nginx/ssl/www.yian.com.crt;
    ssl_certificate_key /etc/nginx/ssl/www.yian.com.key;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;
    ssl_prefer_server_ciphers off;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 1d;

    # Security Headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    add_header Content-Security-Policy "default-src 'self'; script-src 'self' 'unsafe-inline' 'unsafe-eval'; style-src 'self' 'unsafe-inline'; img-src 'self' data: https:; font-src 'self'; connect-src 'self' https:;" always;
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    root /var/www/html/public;
    index index.php;

    charset utf-8;

    # 健康检查
    location /health {
        access_log off;
        return 200 "OK";
        add_header Content-Type text/plain;
    }

    # 静态资源缓存
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
        add_header Vary "Accept-Encoding";
        access_log off;
        try_files $uri =404;
    }

    # 商品图片CDN缓存
    location /storage/products/ {
        expires 7d;
        add_header Cache-Control "public";
        try_files $uri =404;
    }

    # 上传文件（私有）
    location /storage/ {
        internal;
        alias /var/www/html/storage/app/public/;
    }

    # API限流
    location /api/ {
        limit_req zone=api_limit burst=20 nodelay;
        limit_req_status 429;
        add_header X-RateLimit-Limit $limit_req_limit always;
        add_header X-RateLimit-Remaining $limit_req_remaining always;

        proxy_http_version 1.1;
        proxy_set_header Connection "";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-Request-ID $request_id;

        proxy_pass http://octane;
        proxy_read_timeout 300s;
        proxy_send_timeout 300s;
    }

    # Webhook（独立限流）
    location /webhooks/ {
        limit_req zone=webhook_limit burst=50 nodelay;

        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Request-ID $request_id;

        proxy_pass http://octane;
    }

    # 前端路由（SPA）
    location / {
        try_files $uri $uri/ /index.html;
    }

    # 后台路由（SPA）
    location /admin {
        try_files $uri $uri/ /admin/index.html;
    }

    # PHP入口（备用）
    location = /index.php {
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_pass http://octane;
    }

    # 禁止访问敏感文件
    location ~ /\. {
        deny all;
    }
    location ~ \.(env|git|gitignore|gitattributes|lock)$ {
        deny all;
    }
    location ~ ^/(\.env|\.git|\.gitignore|\.gitattributes|composer\.(json|lock))$ {
        deny all;
    }

    # Gzip压缩
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/plain text/css text/xml text/javascript application/json application/javascript application/xml+rss application/rss+xml font/truetype font/opentype application/vnd.ms-fontobject image/svg+xml;
}

# API子域名
server {
    listen 443 ssl http2;
    server_name api.yian.com;

    ssl_certificate /etc/nginx/ssl/api.yian.com.crt;
    ssl_certificate_key /etc/nginx/ssl/api.yian.com.key;

    location / {
        proxy_http_version 1.1;
        proxy_set_header Connection "";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header X-Request-ID $request_id;

        proxy_pass http://octane;
        proxy_read_timeout 300s;
    }
}
```

### 3.2 Nginx限流配置 (nginx.conf http段)

```nginx
http {
    # 限流区域定义
    limit_req_zone $binary_remote_addr zone=api_limit:10m rate=100r/s;
    limit_req_zone $binary_remote_addr zone=login_limit:10m rate=5r/m;
    limit_req_zone $binary_remote_addr zone=webhook_limit:10m rate=50r/s;

    # 连接数限制
    limit_conn_zone $binary_remote_addr zone=conn_limit:10m;

    # 请求ID生成
    map $http_x_request_id $request_id {
        default $http_x_request_id;
        "" $request_id_auto;
    }

    # 日志格式
    log_format main '$remote_addr - $remote_user [$time_local] "$request" '
                    '$status $body_bytes_sent "$http_referer" '
                    '"$http_user_agent" "$http_x_forwarded_for" '
                    'rt=$request_time uct="$upstream_connect_time" '
                    'uht="$upstream_header_time" urt="$upstream_response_time" '
                    'rid="$request_id"';

    access_log /var/log/nginx/access.log main;
    error_log /var/log/nginx/error.log warn;
}
```

## 5. 部署脚本

### 5.1 生产部署脚本 (deploy.sh)

```bash
#!/bin/bash
set -e

# ============================================================
# 怡安印刷商城 — 生产部署脚本
# 用法: ./deploy.sh [staging|production]
# ============================================================

ENV=${1:-staging}
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR="/backup/yian/${TIMESTAMP}"
DEPLOY_DIR="/var/www/yian"
REPO_URL="git@gitlab.com:yian/shop.git"
BRANCH=${ENV}

echo "🚀 开始部署: ${ENV} @ ${TIMESTAMP}"

# 1. 备份当前版本
echo "📦 备份当前版本..."
mkdir -p ${BACKUP_DIR}
cd ${DEPLOY_DIR}
tar czf ${BACKUP_DIR}/code.tar.gz . --exclude=node_modules --exclude=vendor --exclude=storage/logs --exclude=storage/framework/cache
mysqldump -u root -p${DB_PASSWORD} yian_shop > ${BACKUP_DIR}/database.sql
echo "✅ 备份完成: ${BACKUP_DIR}"

# 2. 拉取最新代码
echo "📥 拉取代码..."
cd ${DEPLOY_DIR}
git fetch origin
git checkout ${BRANCH}
git pull origin ${BRANCH}

# 3. 安装依赖
echo "📦 安装PHP依赖..."
composer install --no-dev --optimize-autoloader --no-interaction

echo "📦 安装前端依赖..."
cd resources/js
npm ci
npm run build
cd ../..

# 4. 执行Migration
echo "🗄️ 执行数据库迁移..."
php artisan migrate --force

# 5. 数据填充（仅首次部署）
# php artisan db:seed --force

# 6. 清除缓存
echo "🧹 清除缓存..."
php artisan config:cache
php artisan route:cache
php artisan view:cache
php artisan event:cache

# 7. 预热OPcache
echo "🔥 预热OPcache..."
php artisan octane:reload

# 8. 重启队列
echo "🔄 重启队列..."
php artisan horizon:terminate

# 9. 健康检查
echo "🏥 健康检查..."
sleep 5
HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" https://api.yian.com/health)
if [ "$HTTP_CODE" != "200" ]; then
    echo "❌ 健康检查失败 (HTTP ${HTTP_CODE})，准备回滚..."
    ./rollback.sh ${TIMESTAMP}
    exit 1
fi

echo "✅ 部署成功!"
echo "📊 访问: https://www.yian.com"
echo "📝 备份位置: ${BACKUP_DIR}"
```

### 5.2 回滚脚本 (rollback.sh)

```bash
#!/bin/bash
set -e

TIMESTAMP=$1
BACKUP_DIR="/backup/yian/${TIMESTAMP}"
DEPLOY_DIR="/var/www/yian"

if [ -z "$TIMESTAMP" ]; then
    echo "用法: ./rollback.sh <timestamp>"
    echo "可用备份:"
    ls -la /backup/yian/
    exit 1
fi

echo "⏮️ 开始回滚到: ${TIMESTAMP}"

# 1. 恢复数据库
echo "🗄️ 恢复数据库..."
mysql -u root -p${DB_PASSWORD} yian_shop < ${BACKUP_DIR}/database.sql

# 2. 恢复代码
echo "📦 恢复代码..."
cd ${DEPLOY_DIR}
tar xzf ${BACKUP_DIR}/code.tar.gz

# 3. 清除缓存
php artisan config:cache
php artisan route:cache
php artisan view:cache
php artisan octane:reload
php artisan horizon:terminate

echo "✅ 回滚完成"
```

```bash
#!/bin/bash
# 蓝绿部署：保持旧版本运行，验证新版本后再切换流量

ENV=$1
CURRENT_COLOR=$(cat /var/www/yian/.color 2>/dev/null || echo "green")
NEW_COLOR=$([ "$CURRENT_COLOR" = "green" ] && echo "blue" || echo "green")

echo "当前环境: ${CURRENT_COLOR}"
echo "新版本将部署到: ${NEW_COLOR}"

# 部署到非活跃环境
DEPLOY_DIR="/var/www/yian-${NEW_COLOR}"
# ... 执行部署

# 健康检查
if curl -f https://yian-${NEW_COLOR}.internal/health; then
    # 切换Nginx upstream
    sed -i "s/server app-${CURRENT_COLOR}/server app-${NEW_COLOR}/" /etc/nginx/conf.d/upstream.conf
    nginx -s reload
    
    echo ${NEW_COLOR} > /var/www/yian/.color
    echo "✅ 流量已切换至 ${NEW_COLOR}"
else
    echo "❌ 新版本健康检查失败，保持 ${CURRENT_COLOR} 运行"
    exit 1
fi
```
