## 1. FM工厂端状态机（21状态）

### 1.1 状态定义

| 状态码 | 常量名 | 中文名 | 说明 | nM映射 | 触发角色 |
|:------:|--------|--------|------|:------:|----------|

### 1.3 PHP实现（可直接复制到Enum）

```php
<?php

namespace App\Domains\Orders\Enums;

enum OrderStatus: int
{
    case PendingConfirmation = 0;
    case Confirmed = 1;
    case AwaitingPayment = 3;
    case Paid = 11;
    case AwaitingPrepress = 12;
    case Prepressing = 13;
    case AwaitingPrinting = 15;
    case Printing = 17;
    case AwaitingQC = 20;
    case QCInProgress = 54;
    case AwaitingFinishing = 55;
    case Finishing = 60;
    case AwaitingPackaging = 61;
    case Packaging = 62;
    case AwaitingShipment = 65;
    case Shipped = 66;
    case InTransit = 67;
    case AwaitingReceipt = 69;
    case CancellationRequested = 100;
    case Cancelled = 101;
    case Closed = 255;

    /**
     * 可转移目标状态列表
     */
    public function canTransitionTo(): array
    {
        return match ($this) {
            self::PendingConfirmation => [self::Confirmed, self::CancellationRequested],
            self::Confirmed => [self::AwaitingPayment, self::CancellationRequested],
            self::AwaitingPayment => [self::Paid, self::Cancelled, self::CancellationRequested],
            self::Paid => [self::AwaitingPrepress, self::CancellationRequested],
            self::AwaitingPrepress => [self::Prepressing],
            self::Prepressing => [self::AwaitingPrinting, self::CancellationRequested],
            self::AwaitingPrinting => [self::Printing],
            self::Printing => [self::AwaitingQC],
            self::AwaitingQC => [self::QCInProgress],
            self::QCInProgress => [self::AwaitingFinishing, self::Printing, self::Cancelled],
            self::AwaitingFinishing => [self::Finishing],
            self::Finishing => [self::AwaitingPackaging],
            self::AwaitingPackaging => [self::Packaging],
            self::Packaging => [self::AwaitingShipment],
            self::AwaitingShipment => [self::Shipped],
            self::Shipped => [self::InTransit],
            self::InTransit => [self::AwaitingReceipt],
            self::AwaitingReceipt => [self::Closed],
            self::CancellationRequested => [self::Cancelled], // 驳回时走特殊逻辑恢复
            self::Cancelled => [self::Closed],
            self::Closed => [],
        };
    }

    /**
     * 是否允许从当前状态发起取消申请
     */
    public function canRequestCancel(): bool
    {
        return in_array($this, [
            self::PendingConfirmation,
            self::Confirmed,
            self::AwaitingPayment,
            self::Paid,
            self::Prepressing,
        ], true);
    }

    /**
     * 用户端状态映射（nM）
     */
    public function toCustomerStatus(): CustomerOrderStatus
    {
        return match (true) {
            in_array($this, [self::PendingConfirmation, self::Confirmed, self::AwaitingPayment], true)
                => CustomerOrderStatus::PendingPayment,
            in_array($this, [
                self::Paid, self::AwaitingPrepress, self::Prepressing,
                self::AwaitingPrinting, self::Printing, self::AwaitingQC,
                self::QCInProgress, self::AwaitingFinishing, self::Finishing,
                self::AwaitingPackaging, self::Packaging, self::AwaitingShipment,
            ], true) => CustomerOrderStatus::Producing,
            in_array($this, [self::Shipped, self::InTransit, self::AwaitingReceipt], true)
                => CustomerOrderStatus::AwaitingReceipt,
            $this === self::Cancelled
                => CustomerOrderStatus::Cancelled,
            $this === self::Closed
                => CustomerOrderStatus::Closed,
            default => CustomerOrderStatus::Closed,
        };
    }

    public function label(): string
    {
        return match ($this) {
            self::PendingConfirmation => '待确认',
            self::Confirmed => '已确认',
            self::AwaitingPayment => '待付款',
            self::Paid => '已付款',
            self::AwaitingPrepress => '待印前',
            self::Prepressing => '印前处理中',
            self::AwaitingPrinting => '待印刷',
            self::Printing => '印刷中',
            self::AwaitingQC => '待质检',
            self::QCInProgress => '质检中',
            self::AwaitingFinishing => '待后道',
            self::Finishing => '后道加工中',
            self::AwaitingPackaging => '待打包',
            self::Packaging => '打包中',
            self::AwaitingShipment => '待发货',
            self::Shipped => '已发货',
            self::InTransit => '运输中',
            self::AwaitingReceipt => '待签收',
            self::CancellationRequested => '申请取消中',
            self::Cancelled => '已取消',
            self::Closed => '已关闭',
        };
    }

    /**
     * 状态颜色（前端Tag展示）
     */
    public function color(): string
    {
        return match (true) {
            in_array($this, [self::PendingConfirmation, self::AwaitingPayment], true) => 'warning',
            in_array($this, [self::Confirmed, self::Paid, self::AwaitingPrepress], true) => 'info',
            in_array($this, [
                self::Prepressing, self::AwaitingPrinting, self::Printing,
                self::AwaitingQC, self::QCInProgress, self::AwaitingFinishing,
                self::Finishing, self::AwaitingPackaging, self::Packaging,
                self::AwaitingShipment,
            ], true) => 'primary',
            in_array($this, [self::Shipped, self::InTransit, self::AwaitingReceipt], true) => 'success',
            $this === self::CancellationRequested => 'danger',
            $this === self::Cancelled => 'default',
            $this === self::Closed => 'default',
            default => 'default',
        };
    }
}
```


## 2. nM用户端状态映射（9状态）

### 2.1 状态定义

| 状态码 | 常量名 | 中文名 | 说明 | 对应FM状态 |
|:------:|--------|--------|------|------------|
| `0` | `PendingPayment` | 待付款 | 订单待确认/已确认/待付款 | `0, 1, 3` |
| `1` | `Producing` | 生产中 | 已付款后全部生产流程 | `11~65` |
| `3` | `AwaitingReceipt` | 待收货 | 已发货到待签收 | `66, 67, 69` |
| `4` | `AwaitingReview` | 待评价 | 已签收但未评价（7天内） | `69` + `reviews`为空 |
| `5` | `Completed` | 已完成 | 已签收且已评价或超时7天 | `69` + `reviews`非空 或 超时 |
| `6` | `Cancelled` | 已取消 | 订单已取消 | `101` |
| `7` | `AfterSale` | 售后中 | 已发起售后申请 | `69` + 售后未完结 |
| `8` | `Refunded` | 已退款 | 退款已完成 | `101` + `refund_status=completed` |
| `9` | `Closed` | 已关闭 | 订单归档 | `255` |

### 2.2 PHP实现

```php
<?php

namespace App\Domains\Orders\Enums;

enum CustomerOrderStatus: int
{
    case PendingPayment = 0;
    case Producing = 1;
    case AwaitingReceipt = 3;
    case AwaitingReview = 4;
    case Completed = 5;
    case Cancelled = 6;
    case AfterSale = 7;
    case Refunded = 8;
    case Closed = 9;

    public function label(): string
    {
        return match ($this) {
            self::PendingPayment => '待付款',
            self::Producing => '生产中',
            self::AwaitingReceipt => '待收货',
            self::AwaitingReview => '待评价',
            self::Completed => '已完成',
            self::Cancelled => '已取消',
            self::AfterSale => '售后中',
            self::Refunded => '已退款',
            self::Closed => '已关闭',
        };
    }

    public function color(): string
    {
        return match ($this) {
            self::PendingPayment => 'warning',
            self::Producing => 'primary',
            self::AwaitingReceipt => 'success',
            self::AwaitingReview => 'info',
            self::Completed => 'success',
            self::Cancelled => 'default',
            self::AfterSale => 'danger',
            self::Refunded => 'default',
            self::Closed => 'default',
        };
    }

    /**
     * 从FM状态+业务条件计算nM状态
     */
    public static function fromFmStatus(
        OrderStatus $fm,
        bool $hasReview,
        bool $hasAfterSale,
        bool $refundCompleted,
    ): self {
        return match (true) {
            $fm === OrderStatus::Cancelled && $refundCompleted => self::Refunded,
            $fm === OrderStatus::Cancelled => self::Cancelled,
            $fm === OrderStatus::Closed => self::Closed,
            in_array($fm, [OrderStatus::PendingConfirmation, OrderStatus::Confirmed, OrderStatus::AwaitingPayment], true)
                => self::PendingPayment,
            in_array($fm, [OrderStatus::Shipped, OrderStatus::InTransit, OrderStatus::AwaitingReceipt], true)
                => self::AwaitingReceipt,
            $fm === OrderStatus::AwaitingReceipt && $hasAfterSale => self::AfterSale,
            $fm === OrderStatus::AwaitingReceipt && ! $hasReview => self::AwaitingReview,
            $fm === OrderStatus::AwaitingReceipt && $hasReview => self::Completed,
            default => self::Producing,
        };
    }
}
```

## 3. 售后状态机（6状态 + 3层联动）

### 3.1 三层状态体系

售后业务涉及三个层面的状态联动：
- **EM（External Management）外部管理状态**：客户可见的简化状态
- **AF（AfterSale Flow）售后内部流程状态**：客服/运营处理的详细流程
- **Customer Display**：前端展示的统一状态

### 3.2 状态定义

| 状态码 | EM状态 | AF状态 | Customer Display | 说明 |
|--------|--------|--------|------------------|------|
| `submitted` | 已提交 | `submitted` | 售后申请中 | 客户提交售后申请 |
| `processing` | 处理中 | `awaiting_audit` | 售后审核中 | 等待客服审核 |
| `processing` | 处理中 | `audit_passed` | 审核通过 | 审核通过，等待客户寄回 |
| `processing` | 处理中 | `awaiting_receipt` | 等待收货 | 等待仓库收到退回商品 |
| `processing` | 处理中 | `qc_checking` | 质检中 | 仓库质检退回商品 |
| `completed` | 已完成 | `refunded` | 退款完成 | 退款已到账 |
| `completed` | 已完成 | `reprinted` | 补印完成 | 补印订单已发货 |
| `completed` | 已完成 | `compensated` | 补偿完成 | 优惠货款已补偿 |
| `suspended` | 已挂起 | `awaiting_evidence` | 等待补充凭证 | 客户需补充图片/视频 |
| `revoked` | 已撤销 | `revoked` | 已撤销 | 客户主动撤销 |
| `cancelled` | 已关闭 | `rejected` | 审核驳回 | 客服审核不通过 |

### 3.3 售后类型与流程

| 售后类型 | 代码 | 适用场景 | 标准流程 |
|----------|:----:|----------|----------|
| 退货退款 | `0` | 质量问题/错印/漏印 | submitted → awaiting_audit → audit_passed → awaiting_receipt → qc_checking → refunded |
| 补印 | `1` | 质量问题需重新生产 | submitted → awaiting_audit → audit_passed → reprinted |
| 优惠货款 | `2` | 轻微瑕疵/色差可接受 | submitted → awaiting_audit → audit_passed → compensated |
| 其他 | `3` | 协商一致的特例 | 自定义流程 |

### 3.4 转移矩阵

| 当前AF状态 | 可转移目标 | 触发条件 | 权限 |
|------------|:----------:|----------|------|
| `submitted` | `awaiting_audit` | 系统自动 | 系统 |
| `submitted` | `revoked` | 客户主动撤销（提交后2h内） | 客户 |
| `awaiting_audit` | `audit_passed` | 客服审核通过 | `customer_service` |
| `awaiting_audit` | `rejected` | 客服审核不通过 | `customer_service` |
| `awaiting_audit` | `awaiting_evidence` | 凭证不足需补充 | `customer_service` |
| `awaiting_evidence` | `awaiting_audit` | 客户补充凭证后 | 客户 |
| `audit_passed` | `awaiting_receipt` | 退货退款类型，通知客户寄回 | 系统 |
| `audit_passed` | `reprinted` | 补印类型，创建补印订单 | 系统 |
| `audit_passed` | `compensated` | 优惠货款类型，直接打款 | 系统 |
| `awaiting_receipt` | `qc_checking` | 仓库收到退回商品 | `warehouse` |
| `qc_checking` | `refunded` | 质检通过，执行退款 | `finance` |
| `qc_checking` | `rejected` | 质检不通过（人为损坏/非质量问题） | `warehouse` |
| `rejected` | `closed` | 归档 | 系统（7天后自动） |

```php
<?php

namespace App\Domains\Payments\Services;

use App\Domains\Orders\Models\Order;
use App\Domains\Payments\Enums\RefundPath;

final readonly class RefundApprover
{
    public function __construct(
        private \Illuminate\Contracts\Events\Dispatcher $events,
    ) {}

    #[\NoDiscard]
    public function approve(Order $order, int $amount, string $reason): RefundApprovalResult
    {
        $hoursSinceOrder = now()->diffInHours($order->created_at);
        $isShipped = $order->fm_status->value >= OrderStatus::Shipped->value;
        $isQualityIssue = str_contains($reason, '质量');
        $afterSalePassed = $order->after_sale?->af_status === 'audit_passed';

        // 自动通过规则
        $autoApprove = match (true) {
            ! $isShipped && $amount <= 5000_00 && $hoursSinceOrder < 24 => true,
            $isShipped && ! $isQualityIssue && $amount <= 2000_00 => true,
            $isQualityIssue && $afterSalePassed && $amount <= 1000_00 => true,
            default => false,
        };

        if ($autoApprove) {
            return new RefundApprovalResult(
                approved: true,
                approver: 'system',
                path: $this->determinePath($order, $amount),
            );
        }

        // 需人工审批
        $requiredRole = match (true) {
            $amount > 10000_00 => 'finance_director',
            default => 'customer_service_manager',
        };

        return new RefundApprovalResult(
            approved: false,
            pending: true,
            requiredRole: $requiredRole,
            deadline: now()->addHours($amount > 5000_00 ? 4 : 2),
        );
    }

    private function determinePath(Order $order, int $amount): RefundPath
    {
        return match (true) {
            $order->payment->canRefundOriginal() => RefundPath::Original,
            $order->customer->wallet_balance >= $amount => RefundPath::Wallet,
            $order->customer->bank_card_bound => RefundPath::BankCard,
            default => RefundPath::OfflineTransfer,
        };
    }
}
```

## 7. 计价引擎伪代码

### 7.1 Pipeline 6步计价流程

```php
<?php

namespace App\Domains\Products\Pricing;

use App\Domains\Products\Models\Product;

final readonly class PricingPipeline
{
    /**
     * @param PricingStepInterface[] $steps
     */
    public function __construct(private array $steps) {}

    public function calculate(Product $product, array $params): PricingResult
    {
        $ctx = new PricingContext(product: $product, params: $params);

        foreach ($this->steps as $step) {
            $ctx = $step->apply($ctx);
        }

        return new PricingResult(
            baseAmount: $ctx->baseAmount,
            finalAmount: max($ctx->finalAmount, 50_00), // 最低起订 ¥50
            breakdown: $ctx->breakdown,
        );
    }
}

// ============================================
// Step 1: 基础价格（纸张 + 尺寸 + 数量）
// ============================================
final readonly class BasePriceStep implements PricingStepInterface
{
    public function apply(PricingContext $ctx): PricingContext
    {
        $paper = $ctx->params['paper_type'];      // 如 '157g铜版纸'
        $size = $ctx->params['finished_size'];    // 如 '210x285mm'
        $qty = (int) $ctx->params['quantity'];    // 如 1000
        $color = $ctx->params['color_mode'];      // 如 'cmyk'

        // 纸张单价(元/张) × 印张数 × 色数系数
        $paperPrice = PaperPriceMatrix::lookup($paper, $size); // 如 0.12元/张
        $sheets = ceil($qty / $ctx->product->pieces_per_sheet); // 每版几拼
        $colorFactor = match ($color) {
            'mono' => 1.0,
            'cmyk' => 4.0,
            'cmyk+spot' => 5.0,
            default => 4.0,
        };

        $base = $paperPrice * $sheets * $colorFactor;

        // 放数（印刷损耗）：数量<500 +10%, 500-2000 +8%, 2000-5000 +5%, >5000 +3%
        $wasteRate = match (true) {
            $qty < 500 => 0.10,
            $qty < 2000 => 0.08,
            $qty < 5000 => 0.05,
            default => 0.03,
        };
        $base *= (1 + $wasteRate);

        $ctx->baseAmount = (int) round($base * 100); // 转为分
        $ctx->breakdown['base'] = $ctx->baseAmount;

        return $ctx;
    }
}

// ============================================
// Step 2: 后道工艺加价
// ============================================
final readonly class FinishingStep implements PricingStepInterface
{
    public function apply(PricingContext $ctx): PricingContext
    {
        $processes = $ctx->params['finishing'] ?? []; // ['覆膜', '烫金', '模切']
        $qty = (int) $ctx->params['quantity'];

        $total = 0;
        foreach ($processes as $process) {
            $price = FinishingPriceMatrix::lookup($process, $qty);
            $total += $price;
            $ctx->breakdown["finish_$process"] = $price;
        }

        $ctx->baseAmount += $total;
        return $ctx;
    }
}

// ============================================
// Step 3: 数量折扣
// ============================================
final readonly class QuantityDiscountStep implements PricingStepInterface
{
    public function apply(PricingContext $ctx): PricingContext
    {
        $qty = (int) $ctx->params['quantity'];

        $discountRate = match (true) {
            $qty >= 100000 => 0.45, // 45% off
            $qty >= 50000 => 0.40,
            $qty >= 20000 => 0.35,
            $qty >= 10000 => 0.30,
            $qty >= 5000 => 0.25,
            $qty >= 2000 => 0.20,
            $qty >= 1000 => 0.15,
            $qty >= 500 => 0.10,
            default => 0.0,
        };

        $discount = (int) round($ctx->baseAmount * $discountRate);
        $ctx->baseAmount -= $discount;
        $ctx->breakdown['qty_discount'] = -$discount;

        return $ctx;
    }
}

// ============================================
// Step 4: VIP + 样品折扣
// ============================================
final readonly class VipDiscountStep implements PricingStepInterface
{
    public function apply(PricingContext $ctx): PricingContext
    {
        $customer = $ctx->params['customer'];
        $isSample = (bool) ($ctx->params['is_sample'] ?? false);

        if ($isSample) {
            $discount = (int) round($ctx->baseAmount * 0.50); // 样品5折
            $ctx->baseAmount -= $discount;
            $ctx->breakdown['sample_discount'] = -$discount;
            return $ctx;
        }

        $vipRate = $customer->vipLevel->sampleDiscount(); // 如 Lv5=0.12
        if ($vipRate > 0) {
            $discount = (int) round($ctx->baseAmount * $vipRate);
            $ctx->baseAmount -= $discount;
            $ctx->breakdown['vip_discount'] = -$discount;
        }

        return $ctx;
    }
}

// ============================================
// Step 5: 优惠券抵扣
// ============================================
final readonly class CouponStep implements PricingStepInterface
{
    public function apply(PricingContext $ctx): PricingContext
    {
        $couponCode = $ctx->params['coupon_code'] ?? null;
        if (! $couponCode) {
            return $ctx;
        }

        $coupon = Coupon::findByCode($couponCode);
        if (! $coupon || ! $coupon->isApplicable($ctx)) {
            throw new CouponNotApplicableException($couponCode);
        }

        $discount = $coupon->calculateDiscount($ctx->baseAmount);
        $ctx->baseAmount -= $discount;
        $ctx->breakdown['coupon_discount'] = -$discount;
        $ctx->appliedCoupon = $coupon;

        return $ctx;
    }
}

// ============================================
// Step 6: 加急费 + 包装费 + 0.5元取整
// ============================================
final readonly class UrgencyAndPackagingStep implements PricingStepInterface
{
    public function apply(PricingContext $ctx): PricingContext
    {
        $urgent = (bool) ($ctx->params['is_urgent'] ?? false);
        $qty = (int) $ctx->params['quantity'];

        // 加急费：标准3天，加急1天+50%，2天+30%
        $urgentDays = (int) ($ctx->params['urgent_days'] ?? 0);
        $urgentRate = match ($urgentDays) {
            1 => 0.50,
            2 => 0.30,
            default => 0.0,
        };

        if ($urgentRate > 0) {
            $fee = (int) round($ctx->baseAmount * $urgentRate);
            $ctx->baseAmount += $fee;
            $ctx->breakdown['urgent_fee'] = $fee;
        }

        // 包装费：标准¥2-5/件，VIP Lv5+免
        $packagingFee = PackagingPriceMatrix::lookup(
            $ctx->product->category,
            $qty,
            $ctx->params['customer']->vip_level ?? 0,
        );
        $ctx->baseAmount += $packagingFee;
        $ctx->breakdown['packaging_fee'] = $packagingFee;

        // 0.5元向下取整（如 ¥4623.45 → ¥4623.00；¥4623.55 → ¥4623.00）
        $ctx->finalAmount = (int) (floor($ctx->baseAmount / 50) * 50);
        $ctx->breakdown['floor_adjustment'] = $ctx->finalAmount - $ctx->baseAmount;

        return $ctx;
    }
}
```

```php
<?php

namespace App\Domains\RiskControl\Services;

use App\Domains\Orders\Models\Order;

final readonly class RiskScorer
{
    public function __construct(
        private RiskRuleEngine $ruleEngine,
    ) {}

    public function score(Order $order): RiskScoreResult
    {
        $customer = $order->customer;
        $rules = $this->ruleEngine->evaluate($order, $customer);

        $total = array_sum(array_column($rules, 'score'));
        $total = min($total, 100); // 封顶100

        $level = match (true) {
            $total > 80 => RiskLevel::Block,
            $total > 60 => RiskLevel::ManualReview,
            $total > 40 => RiskLevel::Warning,
            default => RiskLevel::Pass,
        };

        return new RiskScoreResult(
            total: $total,
            level: $level,
            rules: $rules,
            action: $level->action(),
        );
    }
}

enum RiskLevel: int
{
    case Pass = 0;           // 0-40: 直接通过
    case Warning = 1;        // 41-60: 记录日志，正常通过
    case ManualReview = 2;   // 61-80: 进入人工审核队列
    case Block = 3;          // 81-100: 拦截，禁止下单

    public function action(): string
    {
        return match ($this) {
            self::Pass => 'allow',
            self::Warning => 'allow_with_log',
            self::ManualReview => 'manual_review',
            self::Block => 'block',
        };
    }
}
```

```php
<?php

namespace App\Domains\Orders\Enums;

final class OrderModifyStatus
{
    public const int ADDRESS = 1 << 0;       // 1
    public const int QUANTITY = 1 << 1;      // 2
    public const int PAPER = 1 << 2;         // 4
    public const int COLOR = 1 << 3;         // 8
    public const int FINISHING = 1 << 4;     // 16
    public const int DELIVERY_DATE = 1 << 5; // 32
    public const int INVOICE = 1 << 6;       // 64
    public const int FILE = 1 << 7;          // 128

    public const int ALL = self::ADDRESS | self::QUANTITY | self::PAPER
        | self::COLOR | self::FINISHING | self::DELIVERY_DATE
        | self::INVOICE | self::FILE; // 255

    public static function hasFlag(int $status, int $flag): bool
    {
        return ($status & $flag) === $flag;
    }

    public static function addFlag(int &$status, int $flag): void
    {
        $status |= $flag;
    }

    public static function removeFlag(int &$status, int $flag): void
    {
        $status &= ~$flag;
    }

    public static function toArray(int $status): array
    {
        $flags = [];
        foreach ([
            'address' => self::ADDRESS,
            'quantity' => self::QUANTITY,
            'paper' => self::PAPER,
            'color' => self::COLOR,
            'finishing' => self::FINISHING,
            'delivery_date' => self::DELIVERY_DATE,
            'invoice' => self::INVOICE,
            'file' => self::FILE,
        ] as $name => $flag) {
            if (self::hasFlag($status, $flag)) {
                $flags[] = $name;
            }
        }
        return $flags;
    }
}
```

## 11. 积分过期算法

### 11.1 规则

- 积分有效期：**获得之日起12个月**
- 过期检查：每日凌晨 `00:30` Cron 任务执行
- 扣减顺序：**先进先出（FIFO）**，先过期先扣减
- 过期通知：过期前7天、3天、1天各发送一次通知

### 11.2 PHP实现

```php
<?php

namespace App\Domains\Marketing\Services;

use App\Domains\Marketing\Models\PointsTransaction;
use App\Domains\Customers\Models\Customer;

final readonly class PointsExpiryService
{
    public function __construct(
        private \Illuminate\Contracts\Events\Dispatcher $events,
    ) {}

    /**
     * 每日Cron执行：处理过期积分
     */
    public function expireDaily(): void
    {
        $today = now()->startOfDay();

        // 1. 找出今日过期的积分记录（FIFO）
        $expiringRecords = PointsTransaction::query()
            ->where('type', PointsType::Earned->value)
            ->where('expired_at', '<=', $today)
            ->where('remaining', '>', 0)
            ->orderBy('created_at', 'asc')
            ->get();

        foreach ($expiringRecords as $record) {
            $this->expireRecord($record);
        }
    }

    private function expireRecord(PointsTransaction $record): void
    {
        $amount = $record->remaining;
        if ($amount <= 0) {
            return;
        }

        // 2. 创建过期扣减记录
        PointsTransaction::create([
            'customer_id' => $record->customer_id,
            'type' => PointsType::Expired->value,
            'amount' => -$amount,
            'remaining' => 0,
            'source_transaction_id' => $record->id,
            'expired_at' => null,
            'remark' => '积分过期：' . $record->created_at->format('Y-m-d') . '获得',
        ]);

        // 3. 清零原记录
        $record->update(['remaining' => 0]);

        // 4. 触发事件
        $this->events->dispatch(new PointsExpired(
            customerId: $record->customer_id,
            amount: $amount,
        ));
    }

    /**
     * 发送过期预警通知（每日执行）
     */
    public function sendExpiryWarning(): void
    {
        $warnings = [
            7 => '您的积分将在7天后过期，请及时使用',
            3 => '您的积分将在3天后过期，请及时使用',
            1 => '您的积分将在明天过期，请及时使用',
        ];

        foreach ($warnings as $days => $message) {
            $date = now()->addDays($days)->startOfDay();

            $customers = PointsTransaction::query()
                ->where('type', PointsType::Earned->value)
                ->where('expired_at', $date)
                ->where('remaining', '>', 0)
                ->selectRaw('customer_id, SUM(remaining) as total')
                ->groupBy('customer_id')
                ->get();

            foreach ($customers as $customer) {
                NotificationService::send(
                    customerId: $customer->customer_id,
                    channel: NotificationChannel::App,
                    title: '积分即将过期',
                    content: $message . '（过期积分：' . $customer->total . '）',
                );
            }
        }
    }
}
```

## 12. 优惠券计算优先级与互斥规则

### 12.1 优先级（从高到低执行）

1. **免邮券** — 直接减免运费（最高¥30）
2. **直减券** — 固定金额减免（如¥100）
3. **折扣券** — 按比例折扣（如8折），最高减免上限¥500
4. **满减券** — 满X减Y（如满500减50）

### 12.2 互斥规则

| 规则 | 说明 |
|------|------|
| 同类型互斥 | 同一次结算只能使用1张同类型优惠券 |
| 免邮+折扣可叠加 | 免邮券可与折扣券/满减券叠加 |
| 直减与满减互斥 | 直减券和满减券不能同时使用 |
| 折扣与满减互斥 | 折扣券和满减券不能同时使用 |
| VIP折扣与优惠券 | VIP折扣先计算，优惠券在VIP折扣后的金额上计算 |
| 样品订单 | 样品订单（is_sample=true）不能使用任何优惠券 |

### 12.3 计算顺序

```
原始金额
→ 减去 VIP折扣（如有）
→ 减去 数量折扣（Pipeline Step 3已算）
→ 减去 优惠券（按优先级，直减/折扣/满减/免邮）
→ 加上 加急费
→ 加上 包装费
→ 0.5元向下取整
= 最终应付金额
```

## 13. 订单拆分与合并算法

### 13.1 拆分规则

**自动拆分触发条件**：

| 条件 | 拆分策略 |
|------|----------|
| 不同工厂 | 按工厂拆分为多个子订单 |
| 不同发货地 | 按发货仓库拆分为多个子订单 |
| 商品类型差异 | 印刷品与云仓商品拆分 |
| 部分急单 | 急单与非急单拆分 |

**拆分后规则**：
- 子订单号规则：`{父订单号}-S{序号}`（如 `Y2026053000012345-S01`）
- 子订单独立状态机，但共享父订单支付记录
- 子订单可独立发货、独立评价
- 父订单状态 = 所有子订单状态的聚合（最保守原则）

### 13.2 合并规则

**手动合并触发条件**（客户/客服发起）：

| 条件 | 说明 |
|------|------|
| 同客户 | 同一客户账号 |
| 同收货地址 | 收货地址完全一致 |
| 同工厂 | 分配给同一工厂 |
| 同发货批次 | 均未进入印刷中状态 |
| 非急单 | 不含加急订单 |

**合并后规则**：
- 合并订单号：取最早的父订单号
- 被合并订单状态变为 `Merged`
- 合并订单金额 = 各子订单金额之和
- 合并订单交货日期 = 最晚的交货日期

### 13.3 PHP实现

```php
<?php

namespace App\Domains\Orders\Services;

use App\Domains\Orders\Models\Order;
use Illuminate\Support\Collection;

final readonly class OrderSplitter
{
    public function __construct(
        private FactoryAllocator $allocator,
    ) {}

    /**
     * 自动拆分订单
     * @return Order[] 子订单列表
     */
    public function split(Order $order): array
    {
        $groups = $this->groupItems($order->items);

        if ($groups->count() <= 1) {
            return [$order];
        }

        $subOrders = [];
        foreach ($groups as $index => $items) {
            $subOrder = Order::create([
                'parent_id' => $order->id,
                'order_no' => $order->order_no . '-S' . str_pad((string) ($index + 1), 2, '0', STR_PAD_LEFT),
                'customer_id' => $order->customer_id,
                'factory_id' => $items->first()->factory_id,
                'warehouse_id' => $items->first()->warehouse_id,
                'status' => OrderStatus::PendingConfirmation,
                'items' => $items,
                'amount' => $items->sum('amount'),
            ]);
            $subOrders[] = $subOrder;
        }

        $order->update(['is_split' => true, 'split_count' => count($subOrders)]);

        return $subOrders;
    }

    /**
     * 按工厂+发货地+商品类型+急单标志分组
     */
    private function groupItems(Collection $items): Collection
    {
        return $items->groupBy(fn ($item) => implode(':', [
            $item->factory_id,
            $item->warehouse_id,
            $item->product->type->value,
            $item->is_urgent ? 1 : 0,
        ]));
    }
}

final readonly class OrderMerger
{
    /**
     * 手动合并订单
     * @param Order[] $orders 待合并订单（已校验条件）
     */
    public function merge(array $orders): Order
    {
        $baseOrder = collect($orders)->sortBy('created_at')->first();
        $allItems = collect($orders)->flatMap(fn ($o) => $o->items);

        $mergedOrder = Order::create([
            'order_no' => $baseOrder->order_no . '-M',
            'customer_id' => $baseOrder->customer_id,
            'factory_id' => $baseOrder->factory_id,
            'warehouse_id' => $baseOrder->warehouse_id,
            'status' => OrderStatus::PendingConfirmation,
            'items' => $allItems,
            'amount' => $allItems->sum('amount'),
            'deadline' => collect($orders)->max('deadline'),
            'merged_from' => array_map(fn ($o) => $o->id, $orders),
        ]);

        foreach ($orders as $order) {
            $order->update([
                'status' => OrderStatus::Closed,
                'merged_to' => $mergedOrder->id,
                'remark' => '已合并至 ' . $mergedOrder->order_no,
            ]);
        }

        return $mergedOrder;
    }
```
