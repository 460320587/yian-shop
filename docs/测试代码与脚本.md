// cypress/support/commands.ts
declare global {
  namespace Cypress {
    interface Chainable {
      login(phone?: string, password?: string): Chainable<void>;
      loginAdmin(username?: string, password?: string): Chainable<void>;
      clearCart(): Chainable<void>;
    }
  }
}

Cypress.Commands.add('login', (phone = '13800000001', password = 'Test@1234') => {
  cy.session([phone, password], () => {
    cy.request('POST', `${Cypress.env('apiBaseUrl')}/auth/login`, {
      phone,
      password,
      captcha: '1234',
      captcha_key: 'test',
    }).then((response) => {
      window.localStorage.setItem('token', response.body.data.token.access_token);
      window.localStorage.setItem('token_expires_at', String(Math.floor(Date.now() / 1000) + 7200));
    });
  });
});

Cypress.Commands.add('loginAdmin', (username = 'admin001', password = 'Admin@1234') => {
  cy.session([username, password], () => {
    cy.request('POST', `${Cypress.env('adminBaseUrl')}/auth/login`, {
      username,
      password,
      captcha: '1234',
      captcha_key: 'test',
    }).then((response) => {
      window.localStorage.setItem('admin_token', response.body.data.token.access_token);
    });
  });
});

Cypress.Commands.add('clearCart', () => {
  cy.login();
  cy.request({
    method: 'DELETE',
    url: `${Cypress.env('apiBaseUrl')}/cart`,
    headers: { Authorization: `Bearer ${window.localStorage.getItem('token')}` },
  });
});

    $scorer = app(RiskScorer::class);
