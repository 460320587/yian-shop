## 4. Axios实例与拦截器（api/axios.ts）

```typescript
// resources/js/src/api/axios.ts
import axios, { AxiosError, type InternalAxiosRequestConfig } from 'axios';
import { ElMessage, ElNotification } from 'element-plus';
import { useAuthStore } from '@stores/auth';
import router from '@/router';

const api = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || '/api/v1',
  timeout: 30000,
  headers: {
    'Accept': 'application/json',
    'Content-Type': 'application/json',
  },
});

// 请求队列（用于Token刷新时挂起并发请求）
let isRefreshing = false;
let refreshSubscribers: Array<(token: string) => void> = [];

function onRefreshed(token: string) {
  refreshSubscribers.forEach((cb) => cb(token));
  refreshSubscribers = [];
}

function addRefreshSubscriber(cb: (token: string) => void) {
  refreshSubscribers.push(cb);
}

// ====== Request Interceptor ======
api.interceptors.request.use(
  (config: InternalAxiosRequestConfig) => {
    const authStore = useAuthStore();

    // 1. 注入Token（Sanctum SPA模式使用Cookie自动凭证，
    //    但API规范要求部分接口携带Bearer Token，这里兼容两者）
    if (authStore.token) {
      config.headers.Authorization = `Bearer ${authStore.token}`;
    }

    // 2. 幂等Key注入（POST/PUT/DELETE）
    if (['POST', 'PUT', 'DELETE', 'PATCH'].includes(config.method?.toUpperCase() || '')) {
      config.headers['Idempotency-Key'] = generateIdempotencyKey();
    }

    // 3. 请求时间戳（防重放）
    config.headers['X-Request-Time'] = Date.now().toString();

    return config;
  },
  (error) => Promise.reject(error),
);

// ====== Response Interceptor ======
api.interceptors.response.use(
  (response) => {
    // 统一响应结构处理
    const { code, message, data } = response.data;

    if (code !== '0000') {
      ElMessage.error(message || '请求失败');
      return Promise.reject(new Error(message));
    }

    return data;
  },
  async (error: AxiosError<{ code: string; message: string }>) => {
    const { response, config } = error;

    if (!response) {
      ElNotification.error({ title: '网络错误', message: '请检查网络连接后重试' });
      return Promise.reject(error);
    }

    const { status, data } = response;
    const authStore = useAuthStore();

    // 401 Token过期 → 自动刷新
    if (status === 401 && data?.code === '1003' && config && !config.headers?.['X-Retry']) {
      if (!isRefreshing) {
        isRefreshing = true;
        try {
          await authStore.refreshToken();
          isRefreshing = false;
          onRefreshed(authStore.token!);
        } catch {
          isRefreshing = false;
          refreshSubscribers = [];
          authStore.logout();
          router.push('/login');
          return Promise.reject(error);
        }
      }

      // 挂起当前请求，等待刷新完成
      return new Promise((resolve) => {
        addRefreshSubscriber((token: string) => {
          config.headers!.Authorization = `Bearer ${token}`;
          config.headers!['X-Retry'] = '1';
          resolve(api(config));
        });
      });
    }

    // 401 认证失败（非Token过期）
    if (status === 401) {
      ElMessage.error(data?.message || '登录已过期，请重新登录');
      authStore.logout();
      router.push('/login');
      return Promise.reject(error);
    }

    // 403 无权限
    if (status === 403) {
      ElMessage.warning(data?.message || '无权访问该资源');
      return Promise.reject(error);
    }

    // 429 限流
    if (status === 429) {
      ElMessage.warning('操作过于频繁，请稍后再试');
      return Promise.reject(error);
    }

    // 500+ 服务端错误
    if (status >= 500) {
      ElNotification.error({ title: '服务暂不可用', message: '请稍后重试或联系客服' });
      return Promise.reject(error);
    }

    // 其他业务错误
    ElMessage.error(data?.message || `请求失败 (${status})`);
    return Promise.reject(error);
  },
);

// 生成幂等Key（UUIDv7）
function generateIdempotencyKey(): string {
  return crypto.randomUUID ? crypto.randomUUID() : `${Date.now()}-${Math.random().toString(36).slice(2)}`;
}

export default api;
```

### 5.1 Auth Store

```typescript
// resources/js/src/stores/auth.ts
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';
import api from '@api/axios';
import router from '@/router';

export interface User {
  id: number;
  phone: string;
  nickname: string;
  avatar: string;
  customer_type: number;
  vip_level: number;
  company_name?: string;
  auth_status: number;
}

export const useAuthStore = defineStore('auth', () => {
  // State
  const token = ref<string | null>(localStorage.getItem('token'));
  const user = ref<User | null>(null);
  const tokenExpiresAt = ref<number>(parseInt(localStorage.getItem('token_expires_at') || '0'));

  // Getters
  const isLoggedIn = computed(() => !!token.value && tokenExpiresAt.value > Date.now() / 1000);
  const tokenExpiresIn = computed(() => tokenExpiresAt.value - Date.now() / 1000);

  // Actions
  async function login(phone: string, password: string, captcha: string, captchaKey: string) {
    const data = await api.post('/auth/login', { phone, password, captcha, captcha_key: captchaKey });
    setToken(data.token.access_token, data.token.expires_in);
    user.value = data.user;
    return data;
  }

  async function register(payload: RegisterPayload) {
    const data = await api.post('/auth/register', payload);
    setToken(data.token.access_token, data.token.expires_in);
    user.value = data.user;
    return data;
  }

  async function logout() {
    try {
      await api.post('/auth/logout');
    } finally {
      clearToken();
      router.push('/login');
    }
  }

  async function refreshToken() {
    const data = await api.post('/auth/refresh');
    setToken(data.access_token, data.expires_in);
    return data;
  }

  async function fetchUser() {
    const data = await api.get('/customer/me');
    user.value = data;
    return data;
  }

  function setToken(accessToken: string, expiresIn: number) {
    token.value = accessToken;
    tokenExpiresAt.value = Math.floor(Date.now() / 1000) + expiresIn;
    localStorage.setItem('token', accessToken);
    localStorage.setItem('token_expires_at', tokenExpiresAt.value.toString());
  }

  function clearToken() {
    token.value = null;
    user.value = null;
    tokenExpiresAt.value = 0;
    localStorage.removeItem('token');
    localStorage.removeItem('token_expires_at');
  }

  // 初始化时恢复登录状态
  if (token.value && !user.value) {
    fetchUser().catch(() => clearToken());
  }

  return {
    token, user, tokenExpiresAt,
    isLoggedIn, tokenExpiresIn,
    login, register, logout, refreshToken, fetchUser,
  };
});

interface RegisterPayload {
  phone: string;
  password: string;
  sms_code: string;
  captcha: string;
  captcha_key: string;
  invite_code?: string;
}
```

### 5.2 Cart Store

```typescript
// resources/js/src/stores/cart.ts
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';
import api from '@api/axios';

export interface CartItem {
  id: number;
  product_id: number;
  product_name: string;
  product_image: string;
  specs: Record<string, string>;
  quantity: number;
  unit_price: number;
  total_price: number;
  is_selected: boolean;
  is_available: boolean;
  invalid_reason?: string;
}

export const useCartStore = defineStore('cart', () => {
  const items = ref<CartItem[]>([]);
  const loading = ref(false);

  const selectedItems = computed(() => items.value.filter((i) => i.is_selected && i.is_available));
  const selectedTotal = computed(() => selectedItems.value.reduce((sum, i) => sum + i.total_price, 0));
  const selectedCount = computed(() => selectedItems.value.reduce((sum, i) => sum + i.quantity, 0));

  async function fetchCart() {
    loading.value = true;
    try {
      items.value = await api.get('/cart');
    } finally {
      loading.value = false;
    }
  }

  async function addItem(productId: number, specs: Record<string, string>, quantity: number) {
    const data = await api.post('/cart', { product_id: productId, specs, quantity });
    await fetchCart();
    return data;
  }

  async function updateQuantity(itemId: number, quantity: number) {
    await api.put(`/cart/${itemId}`, { quantity });
    const item = items.value.find((i) => i.id === itemId);
    if (item) {
      item.quantity = quantity;
      item.total_price = item.unit_price * quantity;
    }
  }

  async function removeItem(itemId: number) {
    await api.delete(`/cart/${itemId}`);
    items.value = items.value.filter((i) => i.id !== itemId);
  }

  async function toggleSelect(itemId: number) {
    const item = items.value.find((i) => i.id === itemId);
    if (item) {
      item.is_selected = !item.is_selected;
    }
  }

  async function clearCart() {
    await api.delete('/cart');
    items.value = [];
  }

  function selectAll(selected: boolean) {
    items.value.forEach((i) => {
      if (i.is_available) i.is_selected = selected;
    });
  }

  // 初始化加载
  fetchCart();

  return {
    items, loading,
    selectedItems, selectedTotal, selectedCount,
    fetchCart, addItem, updateQuantity, removeItem,
    toggleSelect, clearCart, selectAll,
  };
});
```

```typescript
import api from '@api/axios';

export const useProductStore = defineStore('product', () => {
  const categories = ref<Category[]>([]);
  const products = ref<Product[]>([]);
  const currentProduct = ref<Product | null>(null);
  const loading = ref(false);

  async function fetchCategories() {
    categories.value = await api.get('/categories');
  }

  async function fetchProducts(params?: ProductListParams) {
    loading.value = true;
    try {
      const data = await api.get('/products', { params });
      products.value = data;
      return data;
    } finally {
      loading.value = false;
    }
  }

  async function fetchProductDetail(id: number) {
    loading.value = true;
    try {
      const data = await api.get(`/products/${id}`);
      currentProduct.value = data;
      return data;
    } finally {
      loading.value = false;
    }
  }

  return {
    categories, products, currentProduct, loading,
    fetchCategories, fetchProducts, fetchProductDetail,
  };
});

interface Category {
  id: number;
  name: string;
  icon: string;
  children?: Category[];
}

interface Product {
  id: number;
  name: string;
  subtitle: string;
  images: string[];
  price_range: { min: number; max: number };
}

interface ProductListParams {
  category_id?: number;
  keyword?: string;
  page?: number;
  per_page?: number;
  sort?: string;
}
```

### 5.5 Notification Store

```typescript
// resources/js/src/stores/notification.ts
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';
import api from '@api/axios';

export interface Notification {
  id: number;
  type: string;
  title: string;
  content: string;
  is_read: boolean;
  created_at: string;
  link?: string;
}

export const useNotificationStore = defineStore('notification', () => {
  const notifications = ref<Notification[]>([]);
  const unreadCount = computed(() => notifications.value.filter((n) => !n.is_read).length);

  async function fetchNotifications(params?: { page?: number; per_page?: number }) {
    const data = await api.get('/notifications', { params });
    notifications.value = data;
    return data;
  }

  async function markAsRead(id: number) {
    await api.put(`/notifications/${id}/read`);
    const n = notifications.value.find((i) => i.id === id);
    if (n) n.is_read = true;
  }

  async function markAllAsRead() {
    await api.put('/notifications/read-all');
    notifications.value.forEach((n) => (n.is_read = true));
  }

  async function fetchUnreadCount() {
    const data = await api.get('/notifications/unread-count');
    return data.count;
  }

  return {
    notifications, unreadCount,
    fetchNotifications, markAsRead, markAllAsRead, fetchUnreadCount,
  };
});
```

```typescript
// src/service-worker.ts
const CACHE_NAME = 'yian-shop-v1';
const STATIC_ASSETS = [
  '/',
  '/index.html',
  '/manifest.json',
  '/icon-192.png',
  '/icon-512.png',
];

self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => cache.addAll(STATIC_ASSETS)),
  );
  (self as unknown as ServiceWorkerGlobalScope).skipWaiting();
});

self.addEventListener('activate', (event) => {
  event.waitUntil(
    caches.keys().then((names) =>
      Promise.all(names.filter((n) => n !== CACHE_NAME).map((n) => caches.delete(n))),
    ),
  );
  (self as unknown as ServiceWorkerGlobalScope).clients.claim();
});

self.addEventListener('fetch', (event) => {
  const { request } = event;

  // API请求不缓存
  if (request.url.includes('/api/')) {
    return;
  }

  // 图片缓存7天
  if (request.destination === 'image') {
    event.respondWith(
      caches.match(request).then((cached) => {
        if (cached) return cached;
        return fetch(request).then((response) => {
          const clone = response.clone();
          caches.open(CACHE_NAME).then((cache) => cache.put(request, clone));
          return response;
        });
      }),
    );
    return;
  }

  // 静态资源缓存1年
  event.respondWith(
    caches.match(request).then((cached) => cached || fetch(request)),
  );
});
```

```typescript
// src/composables/usePWA.ts
import { ref, onMounted } from 'vue';

export function usePWA() {
  const deferredPrompt = ref<BeforeInstallPromptEvent | null>(null);
  const canInstall = ref(false);

  onMounted(() => {
    window.addEventListener('beforeinstallprompt', (e) => {
      e.preventDefault();
      deferredPrompt.value = e;
      canInstall.value = true;
    });
  });

  async function install() {
    if (!deferredPrompt.value) return;
    deferredPrompt.value.prompt();
    const { outcome } = await deferredPrompt.value.userChoice;
    if (outcome === 'accepted') {
      canInstall.value = false;
    }
    deferredPrompt.value = null;
  }

  return { canInstall, install };
}
```
