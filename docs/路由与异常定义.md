## 3. 自定义异常类

### 文件路径

### 3.1 基础业务异常

```php
<?php

namespace App\Exceptions;

use Exception;
use Illuminate\Http\JsonResponse;

abstract class BusinessException extends Exception
{
    protected int $httpCode = 400;
    protected string $errorCode = 'BUSINESS_ERROR';
    protected array $data = [];

    public function __construct(
        string $message = '',
        ?string $errorCode = null,
        int $httpCode = 400,
        array $data = []
    ) {
        parent::__construct($message ?: $this->message);
        $this->errorCode = $errorCode ?: $this->errorCode;
        $this->httpCode = $httpCode ?: $this->httpCode;
        $this->data = $data;
    }

    public function getErrorCode(): string
    {
        return $this->errorCode;
    }

    public function getHttpCode(): int
    {
        return $this->httpCode;
    }

    public function getData(): array
    {
        return $this->data;
    }

    public function render(): JsonResponse
    {
        return response()->json([
            'code' => $this->errorCode,
            'message' => $this->getMessage(),
            'data' => $this->data,
        ], $this->httpCode);
    }
}
```

```php
        parent::__construct(
            "操作被风控拦截: {$reason}",
            $this->errorCode,
            403,
            ['risk_score' => $score, 'reason' => $reason]
        );
    }
}
```

## 4. 异常处理器

### 文件路径
`app/Exceptions/Handler.php`

```php
<?php

namespace App\Exceptions;

use Illuminate\Foundation\Exceptions\Handler as ExceptionHandler;
use Illuminate\Validation\ValidationException as LaravelValidationException;
use Illuminate\Database\Eloquent\ModelNotFoundException;
use Illuminate\Auth\AuthenticationException;
use Illuminate\Auth\Access\AuthorizationException;
use Symfony\Component\HttpKernel\Exception\NotFoundHttpException;
use Symfony\Component\HttpKernel\Exception\MethodNotAllowedHttpException;
use Symfony\Component\HttpKernel\Exception\TooManyRequestsHttpException;
use Throwable;

class Handler extends ExceptionHandler
{
    protected $dontReport = [
        //
    ];

    protected $dontFlash = [
        'current_password',
        'password',
        'password_confirmation',
    ];

    public function register(): void
    {
        $this->reportable(function (Throwable $e): void {
            // 记录到Sentry/DingTalk等
            if (app()->isProduction() && $this->shouldReport($e)) {
                // logger()->channel('alert')->error($e);
            }
        });

        $this->renderable(function (BusinessException $e) {
            return $e->render();
        });

        $this->renderable(function (LaravelValidationException $e) {
            return response()->json([
                'code' => 'VALIDATION_ERROR',
                'message' => '参数校验失败',
                'errors' => $e->errors(),
            ], 422);
        });

        $this->renderable(function (ModelNotFoundException $e) {
            return response()->json([
                'code' => 'NOT_FOUND',
                'message' => '资源不存在',
            ], 404);
        });

        $this->renderable(function (AuthenticationException $e) {
            return response()->json([
                'code' => 'UNAUTHENTICATED',
                'message' => '未登录或登录已过期',
            ], 401);
        });

        $this->renderable(function (AuthorizationException $e) {
            return response()->json([
                'code' => 'FORBIDDEN',
                'message' => '无权访问',
            ], 403);
        });

        $this->renderable(function (NotFoundHttpException $e) {
            return response()->json([
                'code' => 'ROUTE_NOT_FOUND',
                'message' => '接口不存在',
            ], 404);
        });

        $this->renderable(function (MethodNotAllowedHttpException $e) {
            return response()->json([
                'code' => 'METHOD_NOT_ALLOWED',
                'message' => '请求方法不允许',
            ], 405);
        });

        $this->renderable(function (TooManyRequestsHttpException $e) {
            return response()->json([
                'code' => 'RATE_LIMIT',
                'message' => '请求过于频繁',
                'retry_after' => $e->getHeaders()['Retry-After'] ?? 60,
            ], 429);
        });
    }

    public function render($request, Throwable $e)
    {
        // 生产环境隐藏详细错误信息
        if (app()->isProduction() && !$e instanceof BusinessException) {
            return response()->json([
                'code' => 'INTERNAL_ERROR',
                'message' => '服务器内部错误',
                'trace_id' => $request->header('X-Trace-Id') ?? uniqid(),
            ], 500);
        }

        return parent::render($request, $e);
    }
}
```
