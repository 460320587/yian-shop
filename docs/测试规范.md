### API-IT-001: 用户注册

```php
it('can register a new user with valid data', function () {
    $response = postJson('/api/v1/auth/register', [
        'phone' => '13800000001',
        'sms_code' => '123456',
        'password' => 'Test@1234',
        'password_confirmation' => 'Test@1234',
        'user_type' => 2,
        'company_name' => '测试公司',
        'credit_code' => '91110108MA00XXXX0X',
        'agreement_version' => 'v2026.05',
    ]);

    $response->assertStatus(201)
        ->assertJsonPath('code', '0000')
        ->assertJsonPath('data.user.phone', '138****0001')
        ->assertJsonStructure([
            'data' => ['user' => ['id', 'phone', 'company_name'], 'tokens']
        ]);

    assertDatabaseHas('users', ['phone' => '13800000001']);
});

it('returns 409 when phone already exists', function () {
    User::factory()->create(['phone' => '13800000001']);
    
    $response = postJson('/api/v1/auth/register', [
        'phone' => '13800000001',
        // ... 其他必填字段
    ]);

    $response->assertStatus(409)
        ->assertJsonPath('code', '1004');
});

it('returns 400 with invalid phone format', function () {
    $response = postJson('/api/v1/auth/register', [
        'phone' => '138000',
    ]);

    $response->assertStatus(400)
        ->assertJsonPath('code', '0001');
});

it('returns 422 with weak password', function () {
    $response = postJson('/api/v1/auth/register', [
        'phone' => '13800000002',
        'password' => '123456',
        // ...
    ]);

    $response->assertStatus(400)
        ->assertJsonValidationErrors('password');
});
```

### API-IT-002: 创建订单（核心交易一致性）

```php
it('creates order with correct amount and inventory deduction', function () {
    $user = User::factory()->create();
    $address = Address::factory()->for($user)->create();
    $product = Product::factory()->create(['min_price' => 2.50]);
    
    // 预置库存
    Redis::set("inventory:{$product->id}", 10000);
    
    $response = actingAs($user)->postJson('/api/v1/orders', [
        'custom_items' => [[
            'product_id' => $product->id,
            'quantity' => 1000,
            'paper_id' => 1,
            'color_id' => 2,
            'pages' => 48,
        ]],
        'address_id' => $address->id,
        'invoice_type' => 0,
    ], ['Idempotency-Key' => (string) Str::uuid7()]);

    $response->assertStatus(201);
    
    $orderNo = $response->json('data.order.order_no');
    
    // 断言1: 订单创建成功
    assertDatabaseHas('orders', [
        'order_no' => $orderNo,
        'customer_id' => $user->id,
    ]);
    
    // 断言2: 库存已预占
    assertDatabaseHas('inventory_reservations', [
        'order_no' => $orderNo,
        'quantity' => 1000,
    ]);
    
    // 断言3: Redis库存扣减
    expect((int) Redis::get("inventory:{$product->id}"))->toBe(9000);
    
    // 断言4: 金额校验
    $order = Order::where('order_no', $orderNo)->first();
    expect($order->total_amount)->toBeGreaterThan(0);
});

it('prevents duplicate order creation with same idempotency key', function () {
    $key = (string) Str::uuid7();
    
    $response1 = actingAs($user)->postJson('/api/v1/orders', $payload, [
        'Idempotency-Key' => $key
    ]);
    
    $response2 = actingAs($user)->postJson('/api/v1/orders', $payload, [
        'Idempotency-Key' => $key
    ]);
    
    $response2->assertStatus(201)
        ->assertJsonPath('data.order.order_no', $response1->json('data.order.order_no'));
    
    // 断言数据库只有1条订单
    assertDatabaseCount('orders', 1);
});

it('returns 409 when inventory insufficient', function () {
    Redis::set("inventory:{$product->id}", 100);
    
    $response = actingAs($user)->postJson('/api/v1/orders', [
        'custom_items' => [[
            'product_id' => $product->id,
            'quantity' => 1000, // 超过库存
        ]],
        // ...
    ]);

    $response->assertStatus(409)
        ->assertJsonPath('code', '4003');
});

it('returns 422 when address does not belong to user', function () {
    $otherAddress = Address::factory()->create(); // 其他用户的地址
    
    $response = actingAs($user)->postJson('/api/v1/orders', [
        'custom_items' => [...],
        'address_id' => $otherAddress->id,
    ]);

    $response->assertStatus(404)
        ->assertJsonPath('code', '6001');
});
```

### API-IT-003: 支付回调幂等性

```php
it('handles duplicate wechat pay callback idempotently', function () {
    $order = Order::factory()->create(['status' => 'pending_payment']);
    $payment = Payment::factory()->for($order)->create(['status' => 'pending']);
    
    $callbackPayload = createWechatCallbackPayload($payment);
    
    // 第一次回调
    $response1 = postJson('/webhooks/wechat-pay', $callbackPayload);
    $response1->assertStatus(200);
    
    // 第二次相同回调
    $response2 = postJson('/webhooks/wechat-pay', $callbackPayload);
    $response2->assertStatus(200);
    
    // 断言订单只被更新一次
    $order->refresh();
    expect($order->status)->toBe('paid');
    
    // 断言只有一个支付成功记录
    assertDatabaseCount('payment_logs', 1);
});
```

## 2. Factory定义

### 2.1 UserFactory

```php
class UserFactory extends Factory
{
    public function definition(): array
    {
        return [
            'phone' => '138' . fake()->unique()->numerify('########'),
            'password_hash' => Hash::make('User@1234'),
            'user_type' => fake()->randomElement([1, 2]),
            'company_id' => null,
            'status' => 1,
            'agreement_version' => 'v2026.05',
            'source' => fake()->randomElement(['web', 'wxh5']),
        ];
    }

    public function company(): static
    {
        return $this->state(fn (array $attributes) => [
            'user_type' => 2,
        ])->afterCreating(function (User $user) {
            $company = Company::factory()->create([
                'auth_status' => 2, // 已认证
            ]);
            $user->update(['company_id' => $company->id]);
        });
    }

    public function vip(int $level = 1): static
    {
        return $this->afterCreating(function (User $user) use ($level) {
            // 设置VIP等级和积分
            $vip = VipLevel::where('level', $level)->first();
            $user->update(['vip_level' => $level]);
            UserProfile::factory()->for($user)->create([
                'points' => $vip->min_points + 100,
            ]);
        });
    }
}
```
